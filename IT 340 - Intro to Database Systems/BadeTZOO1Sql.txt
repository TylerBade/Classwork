Question 7

Question 8

Question 9

SELECT P.PRODUCT_NAME, PP.PRODUCT_PURCHASE_PRICE, RL.SELLING_PRICE, RL.QUANTITY
FROM RECEIPT_LINE RL
JOIN PRODUCT P ON P.PRODUCT_ID = RL.PRODUCT_ID
JOIN PURCHASED_PRODUCT PP ON PP.PRODUCT_ID = P.PRODUCT_ID
WHERE RL.SELLING_PRICE > 3 * PP.PRODUCT_PURCHASE_PRICE;

Question 19

Question 20

Question 21

SELECT (PRODUCT_FINISH_DATE - PRODUCT_START_DATE) AS DAYS_TO_COMPLETE
FROM MANUFACTURED_PRODUCT
WHERE PRODUCT_START_DATE = (SELECT MAX(PRODUCT_START_DATE) FROM MANUFACTURED_PRODUCT);

Question 22

Question 23

SELECT ANIM.ANIM_ID, ANIM.ANIM_NAME
FROM ANIMAL ANIM
JOIN ANIMAL_BREED AB ON ANIM.ANIM_BREED_ID = AB.ANIM_BREED_ID
WHERE AB.ANIM_BREED_NAME = 'Straw-colored Fruit Bat' AND ANIM.ANIM_BIRTH_DATE LIKE '%08';

Question 24

Question 25

Question 26

Question 27

Question 28

SELECT AN.ANIM_NAME, AB.ANIM_BREED_NAME
FROM ANIMAL_BREED AB
JOIN ANIMAL AN ON AN.ANIM_BREED_ID = AB.ANIM_BREED_ID
WHERE (AB.ANIM_BREED_ID BETWEEN 1 AND 5) AND (AN.ANIM_HEIGHT_ON_ARRIVAL > (SELECT AVG(ANIMAL.ANIM_HEIGHT_ON_ARRIVAL) FROM ANIMAL));

Question 29

Question 30
