Get the animal name and breed name of all breeds that may live in tropical or sub-tropical regions.

SELECT ANIM_NAME, ANIM_BREED_NAME
FROM ANIMAL JOIN ANIM_BREED USING (ANIM_BREED_ID)
WHERE ANIM_BREED_HABITAT LIKE ('%Tropcial%') OR ANIM_BREED_HABITAT LIKE ('%sub-tropical%');
