Select the names and suggested prices of products that have a suggested
price UNDER $10, order the names alphabeltically.

SELECT PRODUCT_NAME, PRODUCT_SUGGESTED_PRICE
FROM PRODUCT
WHERE PRODUCT_SUGGESTED_PRICE < 10
ORDER BY PRODUCT_NAME;
